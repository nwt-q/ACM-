            vector<ll>a(n + m + 5);
            ll res = 0;
            for (ll i = 1; i <= n + m + 1; i++) cin >> a[i];
            sort(a.begin() + 1,a.begin() + n + m + 1);
            if(!m) {
                for (ll i = 1; i <= n + m + 1; i++)  res += a[i];
            } else {
                res = a[n + m + 1] - a[1];
                for (ll i = 2; i <= n + m; i++) res += a[i];
            }
            cout << res << endl;
        }
    }
};
