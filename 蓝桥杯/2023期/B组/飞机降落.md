

## [飞机降落](https://www.lanqiao.cn/paper/3818/problem/3511/)

题目考点: `dfs`

```cpp
#include <bits/stdc++.h>
using namespace std;

struct plane {
    int t,d,l;
};

bool vis[15];
bool flag;
std::vector<plane> p(15);

void dfs(int N,int cnt,int last) {
    if (cnt == N) {
        flag = true;
        return;
    }
    for (int i = 0; i < N; i++) {
        if (!vis[i] && p[i].t + p[i].d >= last) {
            vis[i] = true;
            dfs(N, cnt + 1, std::max(last,p[i].t) + p[i].l);
            vis[i] = false;
        }
    }
}

int main()
{
    int T; std::cin >> T;
    for (int i = 0; i < T; i++) {
        int N; std::cin >> N;
        for (int j = 0; j < N; j++) {
            std::cin >> p[j].t >> p[j].d >> p[j].l;
        }
        flag = false;
        dfs(N,0,0);
        if (flag) std::cout << "YES" << "\n";
        else std::cout << "NO" << "\n";
    }
    return 0;
}
```
