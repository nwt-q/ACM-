#include<bits/stdc++.h>

class Solution {
public: //就是一个数的所有公倍数
    void slove() {
        int ans = 0;
        int n,m; cin >> n >> m;
        for (int i = n; i <= m; i++) {
            if (isGood(i)) ans++;
        }
        cout << ans << endl;
    }
    bool isGood(int x) {
        int sum = 0;
        for (int i = 1; i < x; i++) {
            if (x % i == 0)  sum += i;
        }
        if (sum == x) return true;
        else return false;
    }

};
