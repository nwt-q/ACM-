
## [石子合并](https://www.lanqiao.cn/problems/1233/learning/?page=1&first_category_id=1&name=%E7%9F%B3%E5%AD%90)

> 区间dp模版

```cpp
#include<bits/stdc++.h>
#define Run 0
#define endl "\n"
using ll = long long;
using namespace std;

const int MAX = 210;

class Solution {
public: 
    void solve() {
        int n;std::cin >> n;
        std::vector<int> a(n,0);
        memset(dp,127,sizeof dp);
        for (int i = 1; i <= n; i++) {
            std::cin >> a[i];
            a[i] += a[i - 1];
            dp[i][i] = 0; // 长度为 1 的区间不需要合并 代价为 0 
        }
        for (int len = 2; len <= n; len++) {
            for (int i = 1; i + len - 1 <= n; i++) {
                int j = i + len - 1;
                for (int k = i; k < j; k++) {
                    dp[i][j] = min(dp[i][j],dp[i][k] + dp[k + 1][j] + a[j] - a[i - 1]);
                }
            }
        }
        std::cout << dp[1][n] << endl;
    }
private:
    int dp[MAX][MAX];
};

signed main() {
    std::cin.tie(0) -> std::ios::sync_with_stdio(false);
    std::cout.tie(0) -> std::ios::sync_with_stdio(false);
    
#if Run
    int _;std::cin>>_;while(_--) Solution().solve();
#else
    Solution().solve();
#endif
    return 0;
}
```
