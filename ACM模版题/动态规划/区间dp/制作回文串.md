
## [制作回文串](https://www.lanqiao.cn/problems/1547/learning/?page=1&first_category_id=1&name=%E5%88%B6%E4%BD%9C%E5%9B%9E%E6%96%87%E4%B8%B2)

>思路: 讲回文串分区间讨论,环形区间dp

```cpp
#include<bits/stdc++.h>
#define Run 0
#define endl "\n"
#define Pair pair<int,int>
using ll = long long;
using namespace std;


class Solution {
public:
    void solve() {
        int n,m; std::cin >> n >> m;
        std::string s;std::cin >> s;
        std::vector<int>a(26);
        for (int i = 1; i <= m; i++) {
            char t; int x,y; std::cin >> t;std::cin >> x >> y;
           a[t - 'a'] = std::min(x, y);
        }
        std::vector<vector<int>>f(m,vector<int>(m));
        for (int len = 1; len < m; len++) {
        for (int l = 0, r = len; r < m; l++, r++) {
            if (s[l] == s[r])f[l][r] = f[l + 1][r - 1];
            else f[l][r] = std::min(f[l + 1][r] + a[s[l] - 'a'], f[l][r - 1] + a[s[r] - 'a']);
        }
    }
    std::cout << f[0][m - 1] << '\n';
    }
};

signed main() {
    std::cin.tie(0) -> std::ios::sync_with_stdio(false);
    std::cout.tie(0) -> std::ios::sync_with_stdio(false);

#if Run
    int _;std::cin>>_;while(_--) Solution().solve();
#else
    Solution().solve();
#endif
    return 0;
}
```
